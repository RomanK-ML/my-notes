git help - справка

git init - инициализация директории git

git config --global user.name "NAME" -  инициализация имени пользователя
git config --global user.email "EMAIL@email.com" -  инициализация email пользователя

git config --list - просмотр файла конфигурации

git status - показать статус файлов
git add . - добавить все файлы
git commit -a -m"add  file" - сделать коммит

git remote add NAME URL - подключение к проекту GitHub
git remote add ml git@github.com:URL - подключение к проекту через ssh
git push -u NAME main - отправка на сервер GitHub

git clone URL - получить копию проекта
git fetch NAME - получить недостоющие файлы проекта
git pull - загрузить скаченные файлы в проект

git branch NAME - создать ветку
git branch -v - список веток
git chekout NAME - перейти на ветку NAME
git chekout -b NAME - создать и перейти в ветку NAME
